loadData() {
  this.loading = true;
  this.filter.taskNodes = this.taskCodes;

  this.$http.post('schedule:/api/project/ListReceivedProject', {
    data: { filter: this.filter }
  }).then(res => {
    // 处理后端返回的数据
    res.data.forEach(project => {
      project.checked = false;

      // 添加默认的 TS014 任务
      if (!project.PlanNodes.TS014) {
        project.PlanNodes.TS014 = {
          TaskCode: 'TS014',
          TaskName: '新任务',
          StartDate: null,
          EndDate: '默认值'
        };
if (project.PlanNodes.TS013 && project.PlanNodes.TS013.EndDate) {
          const ts013EndDate = new Date(project.PlanNodes.TS013.EndDate);
          ts013EndDate.setDate(ts013EndDate.getDate() + 1);
          project.PlanNodes.TS014.EndDate = ts013EndDate.toISOString();
        }
      }
    });

    this.dataList = res.data;

    if (this.dataList.length > 0) {
      let selected = null;
      if (this.currentProject) {
        selected = this.dataList.find(project => project.ProjectId === this.currentProject.ProjectId);
      }
      if (!selected) {
        selected = this.dataList[0];
      }
      this.$refs.table.setCurrentRow(selected);
      this.$emit('current-change', selected);
    }
  }).finally(() => {
    this.loading = false;
    console.log('After loading data:', this.model.PlanNodes.TS014.EndDate);
  });
},
